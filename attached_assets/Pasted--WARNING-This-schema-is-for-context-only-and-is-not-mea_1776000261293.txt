-- WARNING: This schema is for context only and is not meant to be run.
-- Table order and constraints may not be valid for execution.

CREATE TABLE public.commands (
  id uuid NOT NULL DEFAULT uuid_generate_v4(),
  device_id uuid NOT NULL,
  user_id uuid NOT NULL,
  type text NOT NULL,
  payload jsonb DEFAULT '{}'::jsonb,
  status text NOT NULL DEFAULT 'PENDING'::text CHECK (status = ANY (ARRAY['PENDING'::text, 'EXECUTED'::text, 'FAILED'::text])),
  result jsonb,
  created_at timestamp with time zone DEFAULT now(),
  updated_at timestamp with time zone DEFAULT now(),
  CONSTRAINT commands_pkey PRIMARY KEY (id),
  CONSTRAINT commands_device_id_fkey FOREIGN KEY (device_id) REFERENCES public.devices(id),
  CONSTRAINT commands_user_id_fkey FOREIGN KEY (user_id) REFERENCES auth.users(id)
);
CREATE TABLE public.devices (
  id uuid NOT NULL DEFAULT uuid_generate_v4(),
  user_id uuid NOT NULL,
  device_id text NOT NULL,
  device_name text NOT NULL DEFAULT 'Unknown Device'::text,
  model text DEFAULT ''::text,
  os_version text DEFAULT ''::text,
  manufacturer text DEFAULT ''::text,
  role text NOT NULL DEFAULT 'CLIENT'::text CHECK (role = ANY (ARRAY['OWNER'::text, 'CLIENT'::text])),
  status text NOT NULL DEFAULT 'OFFLINE'::text CHECK (status = ANY (ARRAY['ONLINE'::text, 'OFFLINE'::text])),
  last_seen timestamp with time zone DEFAULT now(),
  created_at timestamp with time zone DEFAULT now(),
  CONSTRAINT devices_pkey PRIMARY KEY (id),
  CONSTRAINT devices_user_id_fkey FOREIGN KEY (user_id) REFERENCES auth.users(id)
);
CREATE TABLE public.logs (
  id uuid NOT NULL DEFAULT uuid_generate_v4(),
  device_id uuid,
  user_id uuid,
  message text NOT NULL,
  level text DEFAULT 'INFO'::text CHECK (level = ANY (ARRAY['INFO'::text, 'WARN'::text, 'ERROR'::text, 'DEBUG'::text])),
  created_at timestamp with time zone DEFAULT now(),
  CONSTRAINT logs_pkey PRIMARY KEY (id),
  CONSTRAINT logs_device_id_fkey FOREIGN KEY (device_id) REFERENCES public.devices(id),
  CONSTRAINT logs_user_id_fkey FOREIGN KEY (user_id) REFERENCES auth.users(id)
);
